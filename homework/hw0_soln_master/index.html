def gutenberg_get_words(url="http://www.gutenberg.org/cache/epub/1232/pg1232.txt", range=slice(0,None), stopwords=[]):
    import requests
    import re
    
    r = requests.get(url)
    data = re.sub(r"[^\w\s]", "", str(r.text)).lower()
   
    return \
        [w for w in data.split() if w not in stopwords]
